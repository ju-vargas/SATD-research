    //que vai ser os estados da unidade de controle!

    in: 
        reset
        end     (no proximo ciclo depois que recebe o end, nao faz mais
                 end deve ser entendido como o sinal que é o último dado indo) 

    out: \\

    regs:
        STATE
        REG_LOAD_HORIZONTAL 


    if (reset):
        REG_LOAD_HORIZONTAL <= 0   
            
        STATE               <= 0
    
    else:
        case(STATE):
        
            state_0:
                -- output --
                REG_LOAD_HORIZONTAL <= not(REG_LOAD_HORIZONTAL)
     
                -- transition --
                //state_0 -> state_1
                STATE <= state_1                
            
            state_1:        
                -- output --
                if(end):
                    REG_LOAD_HORIZONTAL <= not(REG_LOAD_HORIZONTAL)
                
                else: 
                    //continua como ta 
    
                -- transition --
                if(end):    
                    //state_1 -> state_0
                    STATE <= state_0
                    
                else:
                    //continua como ta
                    STATE <= state_1
